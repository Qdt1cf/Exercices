\contenu{ Soit $(u_n)$ la suite d\'efinie pour tout entier naturel $n\geq 1$ par:
$$u_n=\cos(\frac{1}{\sqrt{n}})^{n}-\frac{1}{\sqrt{e}}.$$
\begin{enumerate}
\item \question{Montrer que le d\'eveloppement limit\'e \`a l'ordre $4$ de $x \mapsto \ln(\cos(x))$ en $0$ est:
$$\ln(\cos(x))=-\frac{x^2}{2}-\frac{x^4}{12}+o(x^4).$$
}
\item \question{Montrer qu'au voisinage de $+\infty,$ la suite $(n\ln(\cos(\frac{1}{\sqrt{n}})))_{n \geq 1}$ admet le d\'eveloppement limit\'e suivant:
$$n\ln(\cos(\frac{1}{\sqrt{n}}))=-\frac{1}{2}-\frac{1}{12n}+o(\frac{1}{n}).}$$
\item \question{Donner un \'equivalent de la suite $(u_n)_{n \geq 1}$ et conclure quant \`a la nature de la s\'erie $\displaystyle\sum_{n\geq 1}u_n$.}
\end{enumerate}
