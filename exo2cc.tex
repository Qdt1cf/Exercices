\contenu{ Soit $\alpha$ un nombre r\'eel et $(u_n)$ la suite d\'efinie pour tout entier naturel $n\geq 1$ par:
$$u_n=\left(1+\frac{1}{n^{\alpha}}\right)\times\left(1+\frac{2}{n^{\alpha}}\right)\times \cdots \times \left(1+\frac{n}{n^{\alpha}}\right)-1.$$
\begin{enumerate}
\item \question{Montrer que pour tout entier naturel $n \geq 1,$ la suite $(u_n)$ v\'erifie l'inégalité suivante:
$$u_n \geq \frac{n(n+1)}{2n^{\alpha}}$$

}
\item \question{Donner un équivalent à la suite $(v_n)$ définie par $v_n=\frac{n(n+1)}{2n^{\alpha}}$. En déduire que la série $\displaystyle\sum_{n\geq 1}u_n$ diverge pour $\alpha\leq 3$.}
\item \question{On suppose dans cette question $\alpha>3$. Montrer l'inégalité suivante, pour tout $n \geq 1:$\\
$$0\le u_n \leq (1+\frac{n}{n^{\alpha}})^{n}-1.$$
En déduire que la série $\displaystyle\sum_{n\geq 1}u_n$ converge pour $\alpha>3$.
}
\end{enumerate}
