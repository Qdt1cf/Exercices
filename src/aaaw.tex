\contenu{
\texte{
\begin{enumerate}
\item Déterminer, s'il existe, l'inverse de $3$ modulo $7$.
\item Résoudre l'équation d'inconnue, $x \in \mathbb{Z}\,:\,3x=2 \big[7\big]$.
\item  Résoudre le système d'inconnue, $x \in \mathbb{Z}:$
\[
\begin{cases}
3x \equiv 2 \mod 7 \\
x \equiv 1 \mod 5 \\
x \equiv 5 \mod 12
\end{cases}
\]
\end{enumerate}
}
