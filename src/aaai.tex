\texte{


\begin{enumerate}
    \item Vérifier que \( pgcd(546, 495) = 3 \) et déterminer tous les couples \( (u, v) \in \mathbb{Z}^2 \) tels que
   $
    546u + 495v = 3 ;
  $
    
    \item Déterminer tous les couples \( (x, y) \in \mathbb{Z}^2 \) tels que
$
    546x + 495y = 21 ;
$
    
    \item Déterminer tous les couples \( (x, y) \in \mathbb{Z}^2 \) tels que
  $
    546x + 495y = 22.
   $
\end{enumerate}
}
