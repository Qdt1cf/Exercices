\contenu{
\texte{
On cherche à résoudre l'équation $(E)$ :
$
x^2 + 1 \equiv 0 \mod 65.
$
\begin{enumerate}
    \item Démontrer que \( x \) est solution de l'équation si et seulement si \( x \) est solution de l'un des 4 systèmes :
    \[
   (E_1) 
    \quad  x \equiv 8 \mod 65;  \quad  \,(E_{2})  
    \quad   x \equiv 57 \mod 65 ;
\]
$$ (E_3)
   \begin{cases}
    x \equiv 8 \mod 13 \\
    x \equiv 2 \mod 5
    \end{cases}
    \quad (E_{4})
    \begin{cases}
    x\equiv 5 \mod 13 \\
    x \equiv 3 \mod 5
    \end{cases}
  $$
    \item Déterminer deux entiers \( u \) et \( v \) tels que  $13u + 5v = 1. $
     \item En déduire les solutions de l'équation $(E)$ dans $\mathbb{Z}/65\mathbb{Z} $.
\end{enumerate}


}
