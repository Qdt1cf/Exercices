\titre{ Loi log-normale }
\theme{ probabilités }
\auteur{ Djessy Vianne} 
\organisation{ AMSCC }

\contenu{
    \texte{
    Soit $X$ une variable aléatoire réelle à valeurs dans $\R_+^*$ telle que $\ln(X)$ suive une loi normale d'espérance $\mu = 1$ et d'écart-type $\sigma = 2$. 
    }
    \begin{enumerate}
        \item \question{ Donner une valeur approchée, à $10^{-2}$ près, de $\prob(-0{,}5 \leq \ln(X) \leq 2{,}5)$. }
        \item \question{
            On note $\Phi$ la fonction de répartition de la loi normale centrée réduite $\mathcal{N}(0,1)$ et $F_X$ la fonction de répartition de $X$. Exprimer $F_X$ en fonction de $\Phi$.
        }
        \item \question{
            En déduire la loi de $X$.
        }
    \end{enumerate}

    }
