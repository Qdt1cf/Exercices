\theme { probabilités }
\auteur{ }
\organisation{AMSCC}

\contenu{
\texte{
    Soit $a \in \R$ et $f$ la fonction définie sur $\R$ pour tout $x \in \R$ par :
    $$ f(x) = \begin{cases}
        ax(1-x) & \text{ si } x \in [0,1] \\
        0 & \text{ sinon }
    \end{cases} $$
 }

 \begin{enumerate}
    \item \question{ Déterminer la valeur de $a$ pour que $f$ soit une densité de probabilité. }
    \item \question{ Soit $X$ une variable aléatoire réelle de densité $f$. Déterminer la fonction de répartition $F$ de $X$. }
    \item \question{ Déterminer la probabilité que $X$ prenne une valeur dans l'intervalle $[0,5;1]$. }
    \item \question{ Déterminer l'espérance et la variance de $X$. }
    \item \question{ Déterminer la fonction de répartition de la variable aléatoire $Y = X^2$ et en déduire une densité de probabilité de $Y$. }
 \end{enumerate}
}
