\texte{
\begin{enumerate}
    \item Calculer \( 10^n \mod 11 \), pour tout \( n \in \mathbb{N} \). Déterminer $
    201520142013 \mod 11.
$

    \item En utilisant la congruence \( 10 \equiv -1 \pmod{11} \), déterminer un critère de divisibilité par 11. Appliquer ce critère aux nombres \( 201520142013 \) et \( 201420132012 \).

    \item En utilisant le petit théorème de Fermat, établir que l'entier 
$
    A =201520142013^{2016}-2018^{672}.
 $
    est divisible par $11.$
\end{enumerate}
