\contenu{
Soit l'équation différentielle $(E)$ définie pour $\alpha>0$ par :
\[
\alpha\,y'-(1+x)y=0.
\]
\begin{enumerate}
\item On cherche la solution $f$ sous la forme d'une série entière $f(x)=\displaystyle\sum_{n=0}^{\infty}a_n\,x^{n}$ avec la condition initiale $f(0)=1$. Déterminer les coefficients $(a_n)$ pour $n$ entier et déterminer le rayon de convergence de $f$.
\item Donner une expression alternative de $f,$ solution de $(E),$ avec la condition initiale $f(0)=1$. 
\end{enumerate}

}
