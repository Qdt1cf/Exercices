\organisation{AMSCC}
\contenu{
\texte{
\begin{enumerate}
\item Déterminer le sous groupe multiplicatif engendré par $10$ dans le groupe $\big((\mathbb{Z}/13\mathbb{Z})^{\ast},\times \big)$.
\item Soit la suite $(u_n)_{n \in \mathbb{N}}$ définie par $u_n=10^{n} \pmod{13}.$ Calculer $u_{0},u_1,\cdots,u_{6}$.

\item Calculer $10^{6n+k} \pmod{13}$, pour tout $k \in 0;5:=\{0,1,2,3,4,5\}$.

\item En déduire $10^{2018} \pmod{13}$ et $100^{1000} \pmod{13}$.


\end{enumerate}
}
