\texte{
\begin{enumerate}
\item Démontrer que $10^6 \equiv 1 \pmod 7$.
\item Démontrer que $10^n \equiv 4 \pmod 6$ pour tout $n \in \mathbb{N}^{*}$.
\item En déduire $123456789 \pmod 6$ et que $10^{10^{n}} \equiv 4 \pmod 7$ pour tout $n \in \mathbb{N}^{*}$.
\end{enumerate}
}
